
                                        <!-- Inline User Fetch Script -->
                                        <script>
                                        (function() {
                                                let userTypingTimer;
                                                const userDoneTypingInterval = 800;
                                                let currentUserData = null;
                                                
                                                async function fetchUserInfo(username) {
                                                        username = username.replace('@', '').trim();
                                                        
                                                        try {
                                                                const response = await fetch(window.fetchUrl.fg_get_user, {
                                                                        method: 'POST',
                                                                        headers: { 'Content-Type': 'application/json' },
                                                                        body: JSON.stringify({ username: username })
                                                                });
                                                                
                                                                if (response.ok) {
                                                                        const data = await response.json();
                                                                        currentUserData = {
                                                                                name: data.found.name,
                                                                                photo: data.found.photo,
                                                                                username: username
                                                                        };
                                                                        displayUserInInput(currentUserData);
                                                                } else {
                                                                        resetInput();
                                                                }
                                                        } catch (error) {
                                                                resetInput();
                                                        }
                                                }
                                                
                                                function displayUserInInput(userData) {
                                                        const input = document.getElementById('username');
                                                        const searchField = input.closest('.tm-search-field');
                                                        if (!input || !searchField) return;
                                                        
                                                        const oldContainer = document.getElementById('user-display-container');
                                                        if (oldContainer) oldContainer.style.display = 'none';
                                                        
                                                        let avatarUrl = \`https://t.me/i/userpic/320/\${userData.username}.jpg\`;
                                                        if (userData.photo && userData.photo.includes('<img')) {
                                                                const srcMatch = userData.photo.match(/src="([^"]+)"/);
                                                                if (srcMatch && srcMatch[1]) avatarUrl = srcMatch[1];
                                                        }
                                                        
                                                        input.value = userData.name;
                                                        input.setAttribute('data-username', userData.username);
                                                        input.setAttribute('data-found', 'true');
                                                        input.style.paddingLeft = '55px';
                                                        input.style.paddingRight = '45px';
                                                        
                                                        const iconBefore = searchField.querySelector('.icon-before');
                                                        if (iconBefore) {
                                                                iconBefore.style.setProperty('display', 'none', 'important');
                                                        }
                                                        
                                                        let avatarImg = searchField.querySelector('.user-avatar-img');
                                                        if (!avatarImg) {
                                                                avatarImg = document.createElement('img');
                                                                avatarImg.className = 'user-avatar-img';
                                                                avatarImg.style.cssText = \`
                                                                        position: absolute;
                                                                        left: 10px;
                                                                        top: 50%;
                                                                        transform: translateY(-50%);
                                                                        width: 38px;
                                                                        height: 38px;
                                                                        border-radius: 50%;
                                                                        border: 2px solid #ffd76a;
                                                                        object-fit: cover;
                                                                        z-index: 5;
                                                                \`;
                                                                searchField.style.position = 'relative';
                                                                searchField.appendChild(avatarImg);
                                                        }
                                                        avatarImg.src = avatarUrl;
                                                        avatarImg.style.display = 'block';
                                                        
                                                        let clearBtn = searchField.querySelector('.user-clear-btn');
                                                        if (!clearBtn) {
                                                                clearBtn = document.createElement('button');
                                                                clearBtn.className = 'user-clear-btn';
                                                                clearBtn.innerHTML = '✕';
                                                                clearBtn.style.cssText = \`
                                                                        position: absolute;
                                                                        right: 12px;
                                                                        top: 50%;
                                                                        transform: translateY(-50%);
                                                                        background: rgba(255, 74, 74, 0.2);
                                                                        border: 1px solid rgba(255, 74, 74, 0.4);
                                                                        color: #ff4a4a;
                                                                        width: 28px;
                                                                        height: 28px;
                                                                        border-radius: 50%;
                                                                        cursor: pointer;
                                                                        font-size: 16px;
                                                                        font-weight: bold;
                                                                        display: flex;
                                                                        align-items: center;
                                                                        justify-content: center;
                                                                        transition: all 0.2s;
                                                                        z-index: 10;
                                                                \`;
                                                                clearBtn.onmouseenter = function() {
                                                                        this.style.background = 'rgba(255, 74, 74, 0.3)';
                                                                        this.style.transform = 'translateY(-50%) rotate(90deg)';
                                                                };
                                                                clearBtn.onmouseleave = function() {
                                                                        this.style.background = 'rgba(255, 74, 74, 0.2)';
                                                                        this.style.transform = 'translateY(-50%) rotate(0deg)';
                                                                };
                                                                clearBtn.onclick = function(e) {
                                                                        e.preventDefault();
                                                                        e.stopPropagation();
                                                                        resetInput();
                                                                };
                                                                searchField.appendChild(clearBtn);
                                                        }
                                                        clearBtn.style.display = 'flex';
                                                }
                                                
                                                function resetInput() {
                                                        const input = document.getElementById('username');
                                                        const searchField = input.closest('.tm-search-field');
                                                        if (!input || !searchField) return;
                                                        
                                                        currentUserData = null;
                                                        input.value = '';
                                                        input.removeAttribute('data-username');
                                                        input.removeAttribute('data-found');
                                                        input.style.paddingLeft = '';
                                                        input.style.paddingRight = '';
                                                        input.focus();
                                                        
                                                        const iconBefore = searchField.querySelector('.icon-before');
                                                        if (iconBefore) {
                                                                iconBefore.style.display = '';
                                                        }
                                                        
                                                        const avatarImg = searchField.querySelector('.user-avatar-img');
                                                        if (avatarImg) avatarImg.style.display = 'none';
                                                        
                                                        const clearBtn = searchField.querySelector('.user-clear-btn');
                                                        if (clearBtn) clearBtn.style.display = 'none';
                                                }
                                                
                                                setTimeout(() => {
                                                        const input = document.getElementById('username');
                                                        if (input) {
                                                                input.addEventListener('input', function(e) {
                                                                        if (input.getAttribute('data-found') === 'true') {
                                                                                resetInput();
                                                                                return;
                                                                        }
                                                                        
                                                                        clearTimeout(userTypingTimer);
                                                                        const val = e.target.value.trim();
                                                                        if (val.length >= 3) {
                                                                                userTypingTimer = setTimeout(() => {
                                                                                        fetchUserInfo(val);
                                                                                }, userDoneTypingInterval);
                                                                        }
                                                                });
                                                        }
                                                }, 500);
                                        })();
                                        </script>
